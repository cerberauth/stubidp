Start stubIdP as a background process before your integration tests, then shut it down automatically when the job ends.

## Basic workflow

```yaml
- name: Start stubIdP
  run: |
    npx @cerberauth/stubidp \
      --client-id web-app \
      --client-secret web-app-secret \
      --redirect-uri http://localhost:8080/callback &
    npx wait-on http://localhost:8484/.well-known/openid-configuration

- name: Run integration tests
  run: npm test
  env:
    STUBIDP_ISSUER: http://localhost:8484
    STUBIDP_CLIENT_ID: web-app
    STUBIDP_CLIENT_SECRET: web-app-secret
```

<Aside type="tip">
  Pin explicit client credentials in CI so your test environment configuration stays stable across runs. You can pass
  them as CLI flags (`--client-id`, `--client-secret`, `--redirect-uri`) or as environment variables
  (`STUBIDP_CLIENT_ID`, `STUBIDP_CLIENT_SECRET`, `STUBIDP_REDIRECT_URI`). Both are equivalent — environment variables
  are convenient when the same values are already available as CI secrets.
</Aside>

## Full job example

```yaml
jobs:
  integration:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version: lts/*

      - name: Install dependencies
        run: npm ci

      - name: Start stubIdP
        run: |
          npx @cerberauth/stubidp &
          npx wait-on http://localhost:8484/.well-known/openid-configuration
        env:
          STUBIDP_CLIENT_ID: web-app
          STUBIDP_CLIENT_SECRET: web-app-secret
          STUBIDP_REDIRECT_URI: http://localhost:8080/callback

      - name: Run integration tests
        run: npm test
        env:
          STUBIDP_ISSUER: http://localhost:8484
          STUBIDP_CLIENT_ID: web-app
          STUBIDP_CLIENT_SECRET: web-app-secret
```

## Using a matrix

If you test against multiple redirect URIs or client configurations, pass them through the matrix:

```yaml
strategy:
  matrix:
    redirect_uri:
      - http://localhost:8080/callback
      - http://localhost:9090/auth/callback

steps:
  - name: Start stubIdP
    run: |
      npx @cerberauth/stubidp \
        --client-id web-app \
        --client-secret web-app-secret \
        --redirect-uri ${{ matrix.redirect_uri }} &
      npx wait-on http://localhost:8484/.well-known/openid-configuration
```
